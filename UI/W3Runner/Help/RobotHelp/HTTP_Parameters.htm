Public Property Get Parameters() As

Description:
Return a collection of CVariable. This class is used to stored the parameters name and value.

Sample:

HTTP.URL = "http://1.2.3.4/webservices/WebService.HTTP Demo.asp"
HTTP.Method = "Post"
HTTP.Parameters.Add "Operation" , "Add"
HTTP.Parameters.Add "Param1" , 1
HTTP.Parameters.Add "Param2" , 2
 
If HTTP.Execute()=CHTTP_ERROR_OK Then
 
  strResultVars = Split(HTTP.ResponseText,",")
  m_objReport.AddMessage "Result=" & strResultVars(2)
Else
  W3Runner.TRACE "HTTP call failed " & HTTP.Status & " " & HTTP.StatusText
End If
