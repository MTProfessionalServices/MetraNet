Environ

Public Function Environ(ByVal varKey As Variant) As String
 
' Public Enum eW3RunnerEnvironmentInfo
Public Const w3rW3RUNNER_VERSION = 0
Public Const w3rWEB_BROWSER_VERSION = 1
Public Const w3rWINDOWS_SCRIPT_HOST_VERSION = 2
Public Const w3rVBSCRIPT_ENGINE_VERSION = 3
Public Const w3rVBSCRIPT_DEBUG_MODE = 4
Public Const w3rWINDOWS_VERSION = 5
Public Const w3rCOMPUTER_NAME = 6
Public Const w3rUSER_NAME = 7
Public Const w3rW3RUNNER_PATH = 8
Public Const w3rW3RUNNER_RESERVED_1 = 9
Public Const w3rW3RUNNER_PROGRAM = 10

Description:
Return W3Runner or System enviromnent variables value.

Parameters:

Part

Description

varKey

If varKey is equal to one of the eW3RunnerEnvironmentInfo constants, the function returns the associated value else returns any OS environment variables.

Sample:

If Mid(W3Runner.Environ(w3rWEB_BROWSER_VERSION),1,3)="5.0" Then
  ' With Microsoft Internet Explorer 5.0 we do nothing
Else
  Page.WaitForDownLoad ".asp" ' Wait for the page to refresh
End If
strFileName = W3Runner.Environ(“TEMP”) & “\MyFile.Txt”
