ActiveX

Public Property Get ActiveX() As Variant

Description:

Return the COM instance of an ActiveX. In the case of a MacroMedia Flash object the property return the ShockWave Flash Interface.

Sample:

Page.URL = "W:\Test\ActiveX\Test.htm"
Page.WaitForDownLoad ".htm"
Dim objAXFormLogin
Set objAXFormLogin = Page.Controls("ActiveXLogOn").ActiveX
 
' Accessing the ActiveX Control Interface
objAXFormLogin.TxtLogin.Text = "fsmith"
objAXFormLogin.TxtPassWord.SecureText = "MDEyOS0zMjQwMzE1MjM1NEUzMDQ1NzMxMzcxMTU"
objAXFormLogin.CmdOK.Value = TRUE ' Visual Basic way to raise the click event
 
Page.WaitForDownload "Test.ok.htm"
 
Page.URL = "W:\Test\Flash\MyFlashPage.htm"
Page.WaitForDownLoad ".htm"
Set objFlash = Page.Controls("MyFlash").ActiveX
objFlash.SetVariable "myVar","OK"
objFlash.Play
Do While objFlash.IsPlaying
  W3Runner.Wait 1
Loop
