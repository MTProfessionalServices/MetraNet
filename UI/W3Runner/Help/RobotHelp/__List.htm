List

Public Property Get List(ByVal lngIndex As Long) As String

Description:

Return the text of the lngIndex item. Only supported by ComboBox or ListBox object.

Parameters:

Part

Description

lngIndex

The index of the string to be returned

Sample:

Dim i
 
' Search if an item is in a combobox based on the text
For i = 0 To Page.Controls("cboCountry","ComboBox").ListCount - 1
 
If Page.Controls("cboCountry","ComboBox").List(i)="Italy" Then
 
End If
Next
 
' Search if an item is in a combobox based on the value
For i = 0 To Page.Controls("cboCountry","ComboBox").ListCount - 1
 
If Page.Controls("cboCountry","ComboBox").Value(i)="ITALY" Then
 
End If
Next
