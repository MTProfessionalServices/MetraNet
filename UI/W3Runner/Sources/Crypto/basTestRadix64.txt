' Timer Functions - from Litwin, Getz, Gilbert
'--------------------------------------------------------------
Declare Function wu_GetTime Lib "winmm.dll" Alias _
    "timeGetTime" () As Long
Private mlStartTime As Long
Private Sub ap_StartTimer()
    mlStartTime = wu_GetTime()
End Sub
Private Function ap_EndTimer() As Long
    ap_EndTimer = wu_GetTime() - mlStartTime
End Function
'--------------------------------------------------------------

' NOTE: The functions cv_HexFromString() and cv_StringFromHex()
' are in module basConvert.txt

Public Function testenc64Rand()
    Dim sInput As String, sOutput As String, sDecode As String
    Dim nLen As Integer, i As Integer
    
    ' Fill a string with a random no of random binary values
    Randomize
    sInput = ""
    nLen = Int(32 * Rnd) + 1
    For i = 1 To nLen
       sInput = sInput & Chr$(CByte((Rnd * 256) And &HFF))
    Next
    
    ' Print hex values, encode it, then decode it again
    Debug.Print "Input:", cv_HexFromString(sInput)
    sOutput = EncodeStr64(sInput)
    Debug.Print "Encoded:", sOutput
    sDecode = DecodeStr64(sOutput)
    Debug.Print "Decoded:", cv_HexFromString(sDecode)
    If sInput <> sDecode Then
        MsgBox "Radix64 Error"
    End If

End Function

Public Function TestEnc64Time()
    Dim sInput As String, sOutput As String, sDecode As String
    Dim nLen As Long, i As Long
    Dim lTime As Long
    
    ' Fill a string with a lot of random binary values
    Randomize
    nLen = 100000 + Rnd() * 5
    sInput = String(nLen, " ")
    For i = 1 To nLen
       Mid$(sInput, i, 1) = Chr$(CByte((Rnd * 256) And &HFF))
    Next
    
    ' Encode it, then decode it again
    Call ap_StartTimer
    sOutput = EncodeStr64(sInput)
    sDecode = DecodeStr64(sOutput)
    lTime = ap_EndTimer()
    Debug.Print nLen & " chars took " & lTime & " milliseconds"
    ' Check result equals the input
    If sInput <> sDecode Then
        MsgBox "Radix64 Error"
    End If

End Function
